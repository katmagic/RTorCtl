def initialize(error)
  super(error.desc)

  @error_type = error.secondary_reply_type
  case error.reply_type
    when :permanent_negative_completion then :PERMANENT
    when :temporary_negative_completion then :TEMPORARY
    else :UNKNOWN
  end
end
