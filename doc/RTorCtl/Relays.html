def reload()
  relays = []
  # See doc/spec/dir-spec-v2.txt in Tor's source.
  @rtorctl.getinfo("desc/all-recent").each do |l|
    if l =~ /^router /
      relays << [l]
    else
      relays[-1] << l
    end
  end

  @relays = relays.map{|r| Relay.new(@rtorctl, r)}
end
