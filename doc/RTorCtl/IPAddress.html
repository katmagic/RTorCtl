def []=(index, val)
  if not (0..3) === index
    raise ArgumentError, "there are four parts to an IP address"
  elsif not (0..256) === val
    raise ArgumentError, "each of the IP parts is between 0 and 256"
  end

  ip_as_array = to_a
  ip_as_array[index] = val
  @ip, @netmask = any_to_int_and_netmask(ip_as_array)

  val
end
