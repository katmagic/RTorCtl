def setup
  @rtorctl = RTorCtl::RTorCtl.allocate
  class << @rtorctl
    public :get_response
    attr_accessor :connection

    def write(data)
      @connection.seek(0)
      @connection.write(data)
      @connection.seek(0)
    end
  end

  @rtorctl.connection = StringIO.new
  class << @rtorctl.connection
    alias :old_gets :gets
    def gets(sep="\r\n")
      s = old_gets(sep)
      s and s.chomp(sep)
    end
  end
end
